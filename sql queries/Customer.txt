CREATE TABLE CUSTOMER
(
	CID INT(20) NOT NULL AUTO_INCREMENT,
	NAME VARCHAR(20) NOT NULL,
    PASSWORD VARCHAR(65) NOT NULL,
	EMAIL VARCHAR(20) NOT NULL,
	PHONENO INT(10),
	GENDER VARCHAR(6) NOT NULL,
